# 版本分叉管理设计方案

## 1. 概述

### 1.1 问题描述
当用户回退到某个历史版本后继续编辑更新，会产生版本分叉的情况。传统的线性版本管理无法很好地处理这种场景，需要设计一套完整的分叉版本管理机制。

### 1.2 设计目标
1. **支持版本分叉**：允许从任意历史版本创建新的分支
2. **分支管理**：提供分支的创建、切换、合并、删除功能
3. **冲突处理**：处理分支合并时的内容冲突
4. **版本追溯**：完整的版本历史和分支关系追踪
5. **性能优化**：高效的分支查询和内容重构

## 2. 核心概念

### 2.1 分支（Branch）
- **主分支（Main Branch）**：默认的主要开发分支，通常命名为 `main` 或 `master`
- **特性分支（Feature Branch）**：从某个版本点创建的新分支，用于独立开发
- **分支头（Branch Head）**：分支上的最新版本

### 2.2 版本分叉场景
```
主分支: V1 -> V2 -> V3 -> V4
                |
                +-> V3.1 -> V3.2 (分叉分支)
```

## 4. 核心业务流程

### 4.1 创建分支流程
```java
/**
 * 从指定版本创建新分支
 * @param contentId 内容ID
 * @param fromVersionId 起始版本ID
 * @param branchName 分支名称
 * @param branchType 分支类型
 * @param description 分支描述
 * @param createBy 创建人
 * @return 新分支信息
 */
public ContentBranch createBranch(String contentId, String fromVersionId, 
                                 String branchName, String branchType, 
                                 String description, String createBy) {
    // 1. 验证起始版本存在
    ContentVersion fromVersion = contentVersionMapper.selectById(fromVersionId);
    if (fromVersion == null) {
        throw new BusinessException("起始版本不存在");
    }
    
    // 2. 验证分支名称唯一性
    ContentBranch existingBranch = contentBranchMapper.selectByContentIdAndName(contentId, branchName);
    if (existingBranch != null) {
        throw new BusinessException("分支名称已存在");
    }
    
    // 3. 创建分支记录
    ContentBranch branch = new ContentBranch();
    branch.setId(IdUtil.simpleUUID());
    branch.setContentId(contentId);
    branch.setBranchName(branchName);
    branch.setBranchType(branchType);
    branch.setForkFromVersionId(fromVersionId);
    branch.setHeadVersionId(fromVersionId);
    branch.setDescription(description);
    branch.setCreateBy(createBy);
    
    // 4. 如果是第一个分支，设为默认分支
    long branchCount = contentBranchMapper.countByContentId(contentId);
    if (branchCount == 0) {
        branch.setIsDefault(true);
        branch.setBranchType("MAIN");
    }
    
    contentBranchMapper.insert(branch);
    
    // 5. 更新起始版本的分支归属
    fromVersion.setBranchId(branch.getId());
    contentVersionMapper.updateById(fromVersion);
    
    return branch;
}
```

```java
/**
 * 切换到指定分支
 * @param contentId 内容ID

public ContentVersion switchToBranch(String contentId, String branchId) {
    // 1. 验证分支存在
    ContentBranch branch = contentBranchMapper.selectById(branchId);
    if (branch == null || !branch.getContentId().equals(contentId)) {
        throw new BusinessException("分支不存在");
    }
    
    // 2. 获取分支头版本
    String headVersionId = branch.getHeadVersionId();
    ContentVersion headVersion = contentVersionMapper.selectById(headVersionId);
    
    // 3. 重构版本内容
    return reconstructVersion(headVersion);
}
```

### 4.3 分支更新流程
```java
/**
 * 在当前分支上创建新版本
 * @param contentId 内容ID
 * @param branchId 分支ID
 * @param newContent 新内容
 * @param changeReason 变更原因
 * @param updateBy 更新人
 * @return 新版本信息
 */
public ContentVersion updateBranch(String contentId, String branchId, 
                                  String newContent, String changeReason, String updateBy) {
    // 1. 验证分支存在且激活
    ContentBranch branch = contentBranchMapper.selectById(branchId);
    if (branch == null || !branch.getIsActive()) {
        throw new BusinessException("分支不存在或已停用");
    }
    
    // 2. 获取当前分支头版本
    ContentVersion currentHead = contentVersionMapper.selectById(branch.getHeadVersionId());
    
    // 3. 计算版本号（分支内递增）
    Integer maxVersionNumber = contentVersionMapper.getMaxVersionNumberByBranch(contentId, branchId);
    Integer newVersionNumber = (maxVersionNumber == null ? 1 : maxVersionNumber + 1);
    
    // 4. 创建新版本
    ContentVersion newVersion = createVersionInBranch(contentId, branchId, newVersionNumber, 
                                                     newContent, changeReason, updateBy);
    
    // 5. 更新分支头指针
    branch.setHeadVersionId(newVersion.getId());
    branch.setUpdateBy(updateBy);
    contentBranchMapper.updateById(branch);
    
    // 6. 更新旧版本的is_head标识
    currentHead.setIsHead(false);
    contentVersionMapper.updateById(currentHead);
    
    // 7. 设置新版本为分支头
    newVersion.setIsHead(true);
    contentVersionMapper.updateById(newVersion);
    
    return newVersion;
}
```

### 4.4 分支合并流程
```java
/**
 * 合并分支
 * @param contentId 内容ID
 * @param sourceBranchId 源分支ID
 * @param targetBranchId 目标分支ID
 * @param mergeStrategy 合并策略
 * @param mergeBy 合并人
 * @return 合并结果
 */
@Transactional
public BranchMergeResult mergeBranch(String contentId, String sourceBranchId, 
                                    String targetBranchId, String mergeStrategy, String mergeBy) {
    // 1. 验证分支存在
    ContentBranch sourceBranch = contentBranchMapper.selectById(sourceBranchId);
    ContentBranch targetBranch = contentBranchMapper.selectById(targetBranchId);
    
    if (sourceBranch == null || targetBranch == null) {
        throw new BusinessException("源分支或目标分支不存在");
    }
    
    // 2. 获取分支头版本
    ContentVersion sourceHead = contentVersionMapper.selectById(sourceBranch.getHeadVersionId());
    ContentVersion targetHead = contentVersionMapper.selectById(targetBranch.getHeadVersionId());
    
    // 3. 重构版本内容
    String sourceContent = reconstructVersionContent(sourceHead);
    String targetContent = reconstructVersionContent(targetHead);
    
    // 4. 检测冲突
    MergeConflict conflict = detectMergeConflict(sourceContent, targetContent);
    
    BranchMergeResult result = new BranchMergeResult();
    
    if (conflict.hasConflict() && "AUTO".equals(mergeStrategy)) {
        // 自动合并失败，返回冲突信息
        result.setSuccess(false);
        result.setConflict(conflict);
        return result;
    }
    
    // 5. 执行合并
    String mergedContent;
    if (conflict.hasConflict() && "MANUAL".equals(mergeStrategy)) {
        // 手动解决冲突后的内容
        mergedContent = conflict.getResolvedContent();
    } else {
        // 自动合并
        mergedContent = autoMergeContent(sourceContent, targetContent);
    }
    
    // 6. 在目标分支创建合并版本
    Integer newVersionNumber = contentVersionMapper.getMaxVersionNumberByBranch(contentId, targetBranchId) + 1;
    ContentVersion mergeVersion = createVersionInBranch(contentId, targetBranchId, newVersionNumber,
                                                       mergedContent, "合并分支: " + sourceBranch.getBranchName(), mergeBy);
    
    // 7. 记录合并历史
    ContentBranchMerge mergeRecord = new ContentBranchMerge();
    mergeRecord.setId(IdUtil.simpleUUID());
    mergeRecord.setContentId(contentId);
    mergeRecord.setSourceBranchId(sourceBranchId);
    mergeRecord.setTargetBranchId(targetBranchId);
    mergeRecord.setSourceVersionId(sourceHead.getId());
    mergeRecord.setTargetVersionId(targetHead.getId());
    mergeRecord.setMergeVersionId(mergeVersion.getId());
    mergeRecord.setMergeStrategy(mergeStrategy);
    mergeRecord.setMergeBy(mergeBy);
    
    if (conflict.hasConflict()) {
        mergeRecord.setConflictResolution(conflict.getResolutionDetails());
    }
    
    contentBranchMergeMapper.insert(mergeRecord);
    
    // 8. 更新目标分支头指针
    targetBranch.setHeadVersionId(mergeVersion.getId());
    contentBranchMapper.updateById(targetBranch);
    
    result.setSuccess(true);
    result.setMergeVersion(mergeVersion);
    return result;
}
```

## 5. 冲突检测与解决

### 5.1 冲突检测算法
```java
/**
 * 检测合并冲突
 * @param sourceContent 源分支内容
 * @param targetContent 目标分支内容
 * @return 冲突信息
 */
public MergeConflict detectMergeConflict(String sourceContent, String targetContent) {
    // 使用三路合并算法检测冲突
    // 1. 找到共同祖先版本
    String commonAncestorContent = findCommonAncestor(sourceContent, targetContent);
    
    // 2. 计算差异
    List<DiffEntry> sourceDiffs = DiffUtils.diff(commonAncestorContent, sourceContent);
    List<DiffEntry> targetDiffs = DiffUtils.diff(commonAncestorContent, targetContent);
    
    // 3. 检测冲突区域
    List<ConflictRegion> conflicts = new ArrayList<>();
    
    for (DiffEntry sourceDiff : sourceDiffs) {
        for (DiffEntry targetDiff : targetDiffs) {
            if (isOverlapping(sourceDiff, targetDiff)) {
                ConflictRegion conflict = new ConflictRegion();
                conflict.setStartLine(Math.min(sourceDiff.getStartLine(), targetDiff.getStartLine()));
                conflict.setEndLine(Math.max(sourceDiff.getEndLine(), targetDiff.getEndLine()));
                conflict.setSourceContent(sourceDiff.getContent());
                conflict.setTargetContent(targetDiff.getContent());
                conflicts.add(conflict);
            }
        }
    }
    
    MergeConflict result = new MergeConflict();
    result.setHasConflict(!conflicts.isEmpty());
    result.setConflicts(conflicts);
    
    return result;
}
```

### 5.2 自动合并策略
```java
/**
 * 自动合并内容
 * @param sourceContent 源内容
 * @param targetContent 目标内容
 * @return 合并后内容
 */
public String autoMergeContent(String sourceContent, String targetContent) {
    // 1. 使用三路合并算法
    String commonAncestor = findCommonAncestor(sourceContent, targetContent);
    
    // 2. 应用非冲突的变更
    List<String> sourceLines = Arrays.asList(sourceContent.split("\n"));
    List<String> targetLines = Arrays.asList(targetContent.split("\n"));
    List<String> ancestorLines = Arrays.asList(commonAncestor.split("\n"));
    
    // 3. 使用diff3算法进行三路合并
    List<String> mergedLines = diff3Merge(ancestorLines, sourceLines, targetLines);
    
    return String.join("\n", mergedLines);
}
```

```java
     * 合并分支
     */
    @PostMapping("/{sourceBranchId}/merge/{targetBranchId}")
```

### 8.2 缓存策略
```java
@Service
public class ContentBranchCacheService {
    
    @Autowired
    private RedisTemplate<String, Object> redisTemplate;
    
    private static final String BRANCH_CACHE_KEY = "content:branch:";
    private static final String VERSION_CACHE_KEY = "content:version:";
    private static final int CACHE_EXPIRE_HOURS = 2;
    
    /**
     * 缓存分支信息
     */
    public void cacheBranch(ContentBranch branch) {
        String key = BRANCH_CACHE_KEY + branch.getId();
        redisTemplate.opsForValue().set(key, branch, CACHE_EXPIRE_HOURS, TimeUnit.HOURS);
    }
    
    /**
     * 获取缓存的分支信息
     */
    public ContentBranch getCachedBranch(String branchId) {
        String key = BRANCH_CACHE_KEY + branchId;
        return (ContentBranch) redisTemplate.opsForValue().get(key);
    }
    
    /**
     * 缓存版本内容
     */
    public void cacheVersionContent(String versionId, String content) {
        String key = VERSION_CACHE_KEY + versionId;
        redisTemplate.opsForValue().set(key, content, CACHE_EXPIRE_HOURS, TimeUnit.HOURS);
    }
    
    /**
     * 获取缓存的版本内容
     */
    public String getCachedVersionContent(String versionId) {
        String key = VERSION_CACHE_KEY + versionId;
        return (String) redisTemplate.opsForValue().get(key);
    }
    
    /**
     * 清除分支相关缓存
     */
    public void clearBranchCache(String contentId) {
        String pattern = BRANCH_CACHE_KEY + "*" + contentId + "*";
        Set<String> keys = redisTemplate.keys(pattern);
        if (!keys.isEmpty()) {
            redisTemplate.delete(keys);
        }
    }
}
```

```java
        if (!hasContentAccess(userId, contentId)) {

            case "CREATE_BRANCH":

            case "DELETE_BRANCH":

            default:

        }
    }
```

```java
@Component
public class BranchIntegrityValidator {
    
    /**
     * 验证分支数据完整性
     */
    public void validateBranchIntegrity(String contentId) {
        // 1. 检查默认分支唯一性
        long defaultBranchCount = contentBranchMapper.countDefaultBranches(contentId);
        if (defaultBranchCount != 1) {
            throw new DataIntegrityException("默认分支数量异常: " + defaultBranchCount);
        }
        
        // 2. 检查分支头版本有效性
        List<ContentBranch> branches = contentBranchMapper.selectByContentId(contentId);
        for (ContentBranch branch : branches) {
            ContentVersion headVersion = contentVersionMapper.selectById(branch.getHeadVersionId());
            if (headVersion == null || !headVersion.getBranchId().equals(branch.getId())) {
                throw new DataIntegrityException("分支头版本无效: " + branch.getId());
            }
        }
        
        // 3. 检查版本链完整性
        validateVersionChain(contentId);
    }
    
    /**
     * 验证版本链完整性
     */
    private void validateVersionChain(String contentId) {
        List<ContentVersion> versions = contentVersionMapper.selectByContentId(contentId);
        
        for (ContentVersion version : versions) {
            if (version.getParentSnapshotId() != null) {
                ContentVersion parent = contentVersionMapper.selectById(version.getParentSnapshotId());
                if (parent == null) {
                    throw new DataIntegrityException("父版本不存在: " + version.getParentSnapshotId());
                }
            }
        }
    }
}
```

```java
     */
    public void repairOrphanedBranches() {
        List<ContentBranch> orphanedBranches = contentBranchMapper.findOrphanedBranches();

                // 记录到修复失败列表，人工处理
                recordRepairFailure(branch, e);
            }
        }
    }

        List<ContentVersion> invalidVersions = contentVersionMapper.findInvalidVersions();

            // 检查是否可以安全删除
            if (canSafelyDelete(version)) {
                contentVersionMapper.deleteById(version.getId());
                log.info("清理无效版本: {}", version.getId());
            }
        }
    }
```
