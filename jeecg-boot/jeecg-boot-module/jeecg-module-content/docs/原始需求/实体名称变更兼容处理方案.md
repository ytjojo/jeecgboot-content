# 实体名称变更兼容处理方案

## 概述

在内容社区系统中，当股票名称、用户昵称等实体信息发生变更时，需要优雅地处理历史内容中的引用，确保数据一致性和用户体验。本方案提供了一套完整的兼容处理策略。

## 核心设计理念

1. **ID驱动原则**：所有实体引用基于唯一ID，而非名称
2. **实时同步策略**：实体变更时实时更新相关内容
3. **渐进式更新**：支持批量和增量更新机制
4. **缓存一致性**：确保缓存与数据库的一致性
5. **用户体验优先**：变更过程对用户透明

### 2.1 内容实体更新服务
```java
/**
 * 内容实体更新服务 - 负责更新内容中的实体引用
 */
@Service
@Slf4j
public class ContentEntityUpdateService {
    
    @Autowired
    private ContentEntityReferenceMapper referenceMapper;
    
    @Autowired
    private ContentMapper contentMapper;
    
    @Autowired
    private ContentElementParser contentParser;
    
    /**
     * 更新用户引用
     */
    @Transactional(rollbackFor = Exception.class)
    public UpdateResult updateUserReferences(String userId, String oldNickname, String newNickname) {
        log.info("开始更新用户引用: userId={}, oldName={}, newName={}", userId, oldNickname, newNickname);
        
        // 1. 查找所有相关的内容引用
        List<ContentEntityReference> references = referenceMapper.selectList(
            new LambdaQueryWrapper<ContentEntityReference>()
                .eq(ContentEntityReference::getEntityType, EntityType.USER)
                .eq(ContentEntityReference::getEntityId, userId)
        );
        
        if (references.isEmpty()) {
            log.info("未找到需要更新的用户引用");
            return new UpdateResult(0, 0);
        }
        
        int updatedReferences = 0;
        int updatedContents = 0;
        
        // 2. 按内容ID分组，批量更新
        Map<String, List<ContentEntityReference>> contentGroups = references.stream()
            .collect(Collectors.groupingBy(ContentEntityReference::getContentId));
        
        for (Map.Entry<String, List<ContentEntityReference>> entry : contentGroups.entrySet()) {
            String contentId = entry.getKey();
            List<ContentEntityReference> contentRefs = entry.getValue();
            
            try {
                // 更新单个内容的所有用户引用
                boolean updated = updateContentUserReferences(contentId, contentRefs, oldNickname, newNickname);
                if (updated) {
                    updatedContents++;
                    updatedReferences += contentRefs.size();
                }
            } catch (Exception e) {
                log.error("更新内容{}的用户引用失败", contentId, e);
                // 继续处理其他内容，不中断整个更新过程
            }
        }
        
        log.info("用户引用更新完成: 更新了{}个内容，{}个引用", updatedContents, updatedReferences);
        return new UpdateResult(updatedContents, updatedReferences);
    }
    
    /**
     * 更新股票引用
     */
    @Transactional(rollbackFor = Exception.class)
    public UpdateResult updateStockReferences(String stockId, String oldName, String newName) {
        log.info("开始更新股票引用: stockId={}, oldName={}, newName={}", stockId, oldName, newName);
        
        // 实现逻辑类似updateUserReferences
        List<ContentEntityReference> references = referenceMapper.selectList(
            new LambdaQueryWrapper<ContentEntityReference>()
                .eq(ContentEntityReference::getEntityType, EntityType.STOCK)
                .eq(ContentEntityReference::getEntityId, stockId)
        );
        
        if (references.isEmpty()) {
            return new UpdateResult(0, 0);
        }
        
        int updatedReferences = 0;
        int updatedContents = 0;
        
        Map<String, List<ContentEntityReference>> contentGroups = references.stream()
            .collect(Collectors.groupingBy(ContentEntityReference::getContentId));
        
        for (Map.Entry<String, List<ContentEntityReference>> entry : contentGroups.entrySet()) {
            String contentId = entry.getKey();
            List<ContentEntityReference> contentRefs = entry.getValue();
            
            try {
                boolean updated = updateContentStockReferences(contentId, contentRefs, oldName, newName);
                if (updated) {
                    updatedContents++;
                    updatedReferences += contentRefs.size();
                }
            } catch (Exception e) {
                log.error("更新内容{}的股票引用失败", contentId, e);
            }
        }
        
        return new UpdateResult(updatedContents, updatedReferences);
    }
    
    /**
     * 更新单个内容的用户引用
     */
    private boolean updateContentUserReferences(String contentId, 
                                              List<ContentEntityReference> references, 
                                              String oldNickname, 
                                              String newNickname) {
        // 1. 获取内容
        Content content = contentMapper.selectById(contentId);
        if (content == null) {
            log.warn("内容不存在: {}", contentId);
            return false;
        }
        
        // 2. 更新内容中的用户引用
        String updatedSource = content.getContentSource();
        String updatedHtml = content.getContentHtml();
        boolean hasChanges = false;
        
        for (ContentEntityReference ref : references) {
            // 更新Markdown源码中的引用
            String oldMention = String.format("[@%s](app://community/user/profile?userId=%s)", 
                                             oldNickname, ref.getEntityId());
            String newMention = String.format("[@%s](app://community/user/profile?userId=%s)", 
                                             newNickname, ref.getEntityId());
            
            if (updatedSource.contains(oldMention)) {
                updatedSource = updatedSource.replace(oldMention, newMention);
                hasChanges = true;
            }
            
            // 更新HTML中的引用
            String oldHtmlMention = String.format(
                "<a href=\"app://community/user/profile?userId=%s\" class=\"content-element element-user\" data-type=\"user\" data-id=\"%s\">@%s</a>",
                ref.getEntityId(), ref.getEntityId(), oldNickname);
            String newHtmlMention = String.format(
                "<a href=\"app://community/user/profile?userId=%s\" class=\"content-element element-user\" data-type=\"user\" data-id=\"%s\">@%s</a>",
                ref.getEntityId(), ref.getEntityId(), newNickname);
            
            if (updatedHtml.contains(oldHtmlMention)) {
                updatedHtml = updatedHtml.replace(oldHtmlMention, newHtmlMention);
            }
            
            // 更新引用记录
            ref.setCurrentName(newNickname);
            ref.setUpdatedTime(LocalDateTime.now());
            referenceMapper.updateById(ref);
        }
        
        // 3. 如果有变更，更新内容
        if (hasChanges) {
            content.setContentSource(updatedSource);
            content.setContentHtml(updatedHtml);
            content.setContentVersion(content.getContentVersion() + 1);
            content.setLastUpdated(LocalDateTime.now());
            
            // 重新生成纯文本内容
            content.setContentPlain(contentParser.extractPlainText(updatedSource));
            
            contentMapper.updateById(content);
            return true;
        }
        
        return false;
    }
    
    /**
     * 更新单个内容的股票引用
     */
    private boolean updateContentStockReferences(String contentId, 
                                               List<ContentEntityReference> references, 
                                               String oldName, 
                                               String newName) {
        // 实现逻辑类似updateContentUserReferences
        Content content = contentMapper.selectById(contentId);
        if (content == null) {
            return false;
        }
        
        String updatedSource = content.getContentSource();
        String updatedHtml = content.getContentHtml();
        boolean hasChanges = false;
        
        for (ContentEntityReference ref : references) {
            // 更新Markdown源码中的股票引用
            // 注意：股票引用通常基于代码而非名称，这里主要更新显示文本
            String stockCode = extractStockCode(ref.getEntityId());
            
            // 如果原内容中包含股票名称，则更新
            if (updatedSource.contains(oldName)) {
                updatedSource = updatedSource.replace(oldName, newName);
                hasChanges = true;
            }
            
            // 更新HTML中的股票引用
            if (updatedHtml.contains(oldName)) {
                updatedHtml = updatedHtml.replace(oldName, newName);
            }
            
            // 更新引用记录
            ref.setCurrentName(newName);
            ref.setUpdatedTime(LocalDateTime.now());
            referenceMapper.updateById(ref);
        }
        
        if (hasChanges) {
            content.setContentSource(updatedSource);
            content.setContentHtml(updatedHtml);
            content.setContentVersion(content.getContentVersion() + 1);
            content.setLastUpdated(LocalDateTime.now());
            content.setContentPlain(contentParser.extractPlainText(updatedSource));
            
            contentMapper.updateById(content);
            return true;
        }
        
        return false;
    }
    
    /**
     * 从实体ID中提取股票代码
     */
    private String extractStockCode(String entityId) {
        // 假设entityId格式为: stockCode:market，如 "AAPL:US"
        return entityId.split(":")[0];
    }
}

}
```

### 1.1 缓存层次结构
```java
/**
 * 实体信息缓存管理器
 */
@Component
@Slf4j
public class EntityCacheManager {
    
    @Autowired
    private RedisTemplate<String, Object> redisTemplate;
    
    @Autowired
    private CacheManager localCacheManager;
    
    // 缓存键前缀
    private static final String USER_INFO_PREFIX = "user:info:";
    private static final String STOCK_INFO_PREFIX = "stock:info:";
    private static final String CONTENT_ENTITY_PREFIX = "content:entity:";
    
    // 缓存过期时间
    private static final Duration USER_CACHE_TTL = Duration.ofHours(2);
    private static final Duration STOCK_CACHE_TTL = Duration.ofHours(6);
    private static final Duration CONTENT_CACHE_TTL = Duration.ofMinutes(30);
    
    /**
     * 获取用户信息（多级缓存）
     */
    public UserInfo getUserInfo(String userId) {
        String cacheKey = USER_INFO_PREFIX + userId;
        
        // 1. 先查本地缓存
        Cache localCache = localCacheManager.getCache("user_info");
        UserInfo userInfo = localCache.get(userId, UserInfo.class);
        if (userInfo != null) {
            return userInfo;
        }
        
        // 2. 查Redis缓存
        userInfo = (UserInfo) redisTemplate.opsForValue().get(cacheKey);
        if (userInfo != null) {
            // 回填本地缓存
            localCache.put(userId, userInfo);
            return userInfo;
        }
        
        // 3. 查数据库并缓存
        userInfo = loadUserInfoFromDatabase(userId);
        if (userInfo != null) {
            // 存入Redis
            redisTemplate.opsForValue().set(cacheKey, userInfo, USER_CACHE_TTL);
            // 存入本地缓存
            localCache.put(userId, userInfo);
        }
        
        return userInfo;
    }
    
    /**
     * 获取股票信息（多级缓存）
     */
    public StockInfo getStockInfo(String stockId) {
        String cacheKey = STOCK_INFO_PREFIX + stockId;
        
        // 类似getUserInfo的实现
        Cache localCache = localCacheManager.getCache("stock_info");
        StockInfo stockInfo = localCache.get(stockId, StockInfo.class);
        if (stockInfo != null) {
            return stockInfo;
        }
        
        stockInfo = (StockInfo) redisTemplate.opsForValue().get(cacheKey);
        if (stockInfo != null) {
            localCache.put(stockId, stockInfo);
            return stockInfo;
        }
        
        stockInfo = loadStockInfoFromDatabase(stockId);
        if (stockInfo != null) {
            redisTemplate.opsForValue().set(cacheKey, stockInfo, STOCK_CACHE_TTL);
            localCache.put(stockId, stockInfo);
        }
        
        return stockInfo;
    }
    
    /**
     * 清理用户相关缓存
     */
    public void evictUserCache(String userId) {
        String cacheKey = USER_INFO_PREFIX + userId;
        
        // 清理Redis缓存
        redisTemplate.delete(cacheKey);
        
        // 清理本地缓存
        Cache localCache = localCacheManager.getCache("user_info");
        localCache.evict(userId);
        
        // 发送缓存失效通知给其他节点
        sendCacheEvictionNotification("user", userId);
        
        log.info("已清理用户缓存: {}", userId);
    }
    
    /**
     * 清理股票相关缓存
     */
    public void evictStockCache(String stockId) {
        String cacheKey = STOCK_INFO_PREFIX + stockId;
        
        redisTemplate.delete(cacheKey);
        
        Cache localCache = localCacheManager.getCache("stock_info");
        localCache.evict(stockId);
        
        sendCacheEvictionNotification("stock", stockId);
        
        log.info("已清理股票缓存: {}", stockId);
    }
    
    /**
     * 发送缓存失效通知
     */
    private void sendCacheEvictionNotification(String entityType, String entityId) {
        CacheEvictionMessage message = new CacheEvictionMessage(entityType, entityId, System.currentTimeMillis());
        
        // 通过消息队列通知其他节点
        redisTemplate.convertAndSend("cache:eviction", message);
    }
    
    /**
     * 从数据库加载用户信息
     */
    private UserInfo loadUserInfoFromDatabase(String userId) {
        // 实现数据库查询逻辑
        return userMapper.selectUserInfoById(userId);
    }
    
    /**
     * 从数据库加载股票信息
     */
    private StockInfo loadStockInfoFromDatabase(String stockId) {
        // 实现数据库查询逻辑
        return stockMapper.selectStockInfoById(stockId);
    }
}
```

```java
                case "user":
                    evictLocalUserCache(message.getEntityId());
```

```java
/**
 * 批量实体更新任务
 */
@Component
@Slf4j
public class BatchEntityUpdateTask {
    
    @Autowired
    private ContentEntityUpdateService updateService;
    
    @Autowired
    private RedisTemplate<String, Object> redisTemplate;
    
    // 批量更新队列键
    private static final String BATCH_UPDATE_QUEUE = "batch:entity:update";
    
    /**
     * 添加批量更新任务
     */
    public void addBatchUpdateTask(EntityUpdateTask task) {
        redisTemplate.opsForList().rightPush(BATCH_UPDATE_QUEUE, task);
        log.info("已添加批量更新任务: {}", task);
    }
    
    /**
     * 定时处理批量更新任务
     */
    @Scheduled(fixedDelay = 30000) // 每30秒执行一次
    public void processBatchUpdateTasks() {
        try {
            List<Object> tasks = redisTemplate.opsForList().range(BATCH_UPDATE_QUEUE, 0, 99); // 每次最多处理100个任务
            if (tasks == null || tasks.isEmpty()) {
                return;
            }
            
            log.info("开始处理{}个批量更新任务", tasks.size());
            
            // 按实体类型分组
            Map<EntityType, List<EntityUpdateTask>> groupedTasks = tasks.stream()
                .map(obj -> (EntityUpdateTask) obj)
                .collect(Collectors.groupingBy(EntityUpdateTask::getEntityType));
            
            // 分别处理不同类型的实体更新
            for (Map.Entry<EntityType, List<EntityUpdateTask>> entry : groupedTasks.entrySet()) {
                EntityType entityType = entry.getKey();
                List<EntityUpdateTask> entityTasks = entry.getValue();
                
                processBatchUpdateByType(entityType, entityTasks);
            }
            
            // 清理已处理的任务
            redisTemplate.opsForList().trim(BATCH_UPDATE_QUEUE, tasks.size(), -1);
            
        } catch (Exception e) {
            log.error("处理批量更新任务失败", e);
        }
    }
    
    /**
     * 按类型处理批量更新
     */
    private void processBatchUpdateByType(EntityType entityType, List<EntityUpdateTask> tasks) {
        log.info("处理{}类型的{}个更新任务", entityType, tasks.size());
        
        switch (entityType) {
            case USER:
                processBatchUserUpdates(tasks);
                break;
            case STOCK:
                processBatchStockUpdates(tasks);
                break;
            default:
                log.warn("不支持的实体类型: {}", entityType);
        }
    }
    
    /**
     * 批量处理用户更新
     */
    private void processBatchUserUpdates(List<EntityUpdateTask> tasks) {
        for (EntityUpdateTask task : tasks) {
            try {
                UpdateResult result = updateService.updateUserReferences(
                    task.getEntityId(),
                    task.getOldName(),
                    task.getNewName()
                );
                
                log.info("用户批量更新完成: userId={}, 更新了{}个内容", 
                        task.getEntityId(), result.getUpdatedContents());
                
            } catch (Exception e) {
                log.error("批量更新用户{}失败", task.getEntityId(), e);
            }
        }
    }
    
    /**
     * 批量处理股票更新
     */
    private void processBatchStockUpdates(List<EntityUpdateTask> tasks) {
        for (EntityUpdateTask task : tasks) {
            try {
                UpdateResult result = updateService.updateStockReferences(
                    task.getEntityId(),
                    task.getOldName(),
                    task.getNewName()
                );
                
                log.info("股票批量更新完成: stockId={}, 更新了{}个内容", 
                        task.getEntityId(), result.getUpdatedContents());
                
            } catch (Exception e) {
                log.error("批量更新股票{}失败", task.getEntityId(), e);
            }
        }
    }
}
```

2. **外键关系**：通过应用层维护外键关系，避免数据库层面的强约束
3. **索引策略**：
   - 复合索引优先：`(entity_type, entity_id)` 支持按实体类型和ID查询
   - 位置索引：`(content_id, position_start)` 支持按内容和位置排序
   - 时间索引：支持按时间范围查询和清理历史数据
4. **版本控制**：通过version字段实现乐观锁和变更检测
5. **软删除**：重要数据表支持软删除，避免误删除

```java
/**
 * 失败更新补偿任务
 */
@Component
@Slf4j
public class FailedUpdateCompensationTask {
    
    @Autowired
    private FailedUpdateRecordService failedUpdateRecordService;
    
    @Autowired
    private ReliableEntityUpdateService reliableUpdateService;
    
    /**
     * 定时处理失败的更新
     */
    @Scheduled(fixedDelay = 300000) // 每5分钟执行一次
    public void processFailedUpdates() {
        try {
            List<FailedUpdateRecord> failedRecords = failedUpdateRecordService.getPendingRecords(50);
            if (failedRecords.isEmpty()) {
                return;
            }
            
            log.info("开始处理{}个失败的更新记录", failedRecords.size());
            
            for (FailedUpdateRecord record : failedRecords) {
                processFailedRecord(record);
            }
            
        } catch (Exception e) {
            log.error("处理失败更新记录时出错", e);
        }
    }
    
    /**
     * 处理单个失败记录
     */
    private void processFailedRecord(FailedUpdateRecord record) {
        // 检查重试次数限制
        if (record.getRetryCount() >= 5) {
            log.warn("更新记录重试次数超限，标记为放弃: {}", record.getId());
            record.setStatus(FailedUpdateStatus.ABANDONED);
            record.setResolvedTime(LocalDateTime.now());
            failedUpdateRecordService.update(record);
            return;
        }
        
        try {
            // 更新状态为重试中
            record.setStatus(FailedUpdateStatus.RETRYING);
            record.setRetryCount(record.getRetryCount() + 1);
            record.setLastRetryTime(LocalDateTime.now());
            failedUpdateRecordService.update(record);
            
            // 执行重试
            UpdateResult result;
            if (record.getEntityType() == EntityType.USER) {
                result = reliableUpdateService.updateUserReferencesReliably(
                    record.getEntityId(),
                    record.getOldName(),
                    record.getNewName()
                );
            } else if (record.getEntityType() == EntityType.STOCK) {
                result = reliableUpdateService.updateStockReferencesReliably(
                    record.getEntityId(),
                    record.getOldName(),
                    record.getNewName()
                );
            } else {
                log.warn("不支持的实体类型: {}", record.getEntityType());
                return;
            }
            
            // 重试成功，标记为已解决
            record.setStatus(FailedUpdateStatus.RESOLVED);
            record.setResolvedTime(LocalDateTime.now());
            failedUpdateRecordService.update(record);
            
            log.info("失败更新记录重试成功: id={}, 更新了{}个内容", 
                    record.getId(), result.getUpdatedContents());
            
        } catch (Exception e) {
            log.error("重试失败更新记录失败: id={}", record.getId(), e);
            
            // 重试失败，恢复为待处理状态
            record.setStatus(FailedUpdateStatus.PENDING);
            failedUpdateRecordService.update(record);
        }
    }
}
```

```java
        ConsistencyCheckResult result = new ConsistencyCheckResult();
        result.setCheckType("USER_REFERENCE");
        result.setStartTime(LocalDateTime.now());
        
        try {
            // 查找所有用户引用
            List<ContentEntityReference> userReferences = referenceMapper.selectList(
                new LambdaQueryWrapper<ContentEntityReference>()
                    .eq(ContentEntityReference::getEntityType, EntityType.USER)
            );

            int inconsistentCount = 0;
            List<String> inconsistentItems = new ArrayList<>();
            
            for (ContentEntityReference ref : userReferences) {
                // 检查用户是否存在
                User user = userMapper.selectById(ref.getEntityId());
                if (user == null) {
                    inconsistentCount++;
                    inconsistentItems.add(String.format("用户不存在: userId=%s, contentId=%s",

                                                       ref.getEntityId(), ref.getCurrentName(), user.getNickname()));
                }

            result.setInconsistentCount(inconsistentCount);
```

2. **多级缓存策略**：本地缓存+分布式缓存，优化性能
3. **异步处理机制**：批量更新+异步通知，提升系统响应
4. **智能重试策略**：指数退避+失败补偿，提高成功率
5. **实时一致性检查**：主动发现和修复数据不一致问题

通过这套方案，可以优雅地处理股票名称、用户昵称等实体信息的变更，确保历史内容中的引用能够及时、准确地更新，为用户提供一致的体验。
